use std::path::PathBuf;
use walkdir::WalkDir;

#[derive(Debug, Clone)]
pub enum Severity {
    Critical, High, Medium, Low,
}
#[derive(Debug)]
pub struct FindingVuln {
    pub file:     PathBuf,
    pub line:     u32,
    pub message:  String,
    pub severity: Severity,
}
pub trait Scannable {
    fn scan(&self) -> Vec<FindingVuln>;
    fn is_ok(&self, ext: &str) -> bool;
}
const MAX: u64 = 10_485_760;
const RE:  &str = r"LDAP://[\w\.]+";
pub fn scan_file<'a>(
    path: &'a PathBuf,
    pats: &[(&str, Severity)],
) -> Vec<FindingVuln> {
    let text = std::fs::read_to_string(path).unwrap_or_default();
    let mut out: Vec<FindingVuln> = Vec::new();
    for (i, ln) in text.lines().enumerate() {
        for (p, s) in pats {
            if ln.contains(p) {
                out.push(FindingVuln {
                    file:     path.clone(),
                    line:     i as u32 + 1,
                    message:  format!("match:{p}"),
                    severity: s.clone(),
                });
            }
        }
    }
    out
}

from dataclasses import dataclass, field
from pathlib    import Path
import re

MAX     = 500
LDAP_RE = re.compile(r"LDAP://[\w\.]+")

class Severity:
    CRITICAL = "critical"
    HIGH     = "high"
    MEDIUM   = "medium"
    LOW      = "low"

@dataclass
class ScanResult:
    file_path: Path
    line_no:   int
    pattern:  str
    severity: str  = "medium"
    hits:     list = field(default_factory=list)

def scan(path:Path, rules:list) -> list:
    """Scan file against all rules."""
    if not path.exists(): return []
    out, text = [], path.read_text()
    for n, line in enumerate(text.splitlines(), 1):
        for r in rules:
            if re.search(r.regex,line):
                out.append(ScanResult(
                    file_path=path, line_no=n,
                    pattern=r.name, severity=r.severity,
                ))
    return out[:MAX]

# Install-Package.ps1 -- winget wrapper
function Install-Package {
    [CmdletBinding()]
    param (
        [Parameter(Mandatory=$true)] [string] $Query,
        [Parameter()]               [switch] $Uninstall
    )
    $raw = winget search --query $Query
    if (-not $raw) { Write-Warning "No results"; return }
    $pkgs = $raw | Select-Object -Skip 2 | ForEach-Object {
        $c = $_ -split '\s{2,}', 4
        [PSCustomObject] @{
            Name = $c[0]; Id = $c[1]; Ver = $c[2]
        }
    }
    $pkgs | Format-Table -AutoSize
    $i = [int](Read-Host "Pick") - 1
    if ($Uninstall) { winget uninstall --id $pkgs[$i].Id
    } else { winget install --id $pkgs[$i].Id }
}

// scanner.go -- Go edition
package scanner
import ( "bufio"; "fmt"; "os"; "regexp" )
var RE = regexp.MustCompile(`LDAP://[\w\.]+`)
type Severity string
const (
    Critical Severity = "critical"
    High     Severity = "high"
    Medium   Severity = "medium"
    Low      Severity = "low"
)
type FindingVuln struct {
    File,Message string
    Line        int
    Severity    Severity
}
func ScanFile(path string, pats []*regexp.Regexp) ([]FindingVuln, error) {
    f,err := os.Open(path)
    if err != nil { return nil,err }
    defer f.Close()
    var out []FindingVuln
    sc,n := bufio.NewScanner(f),0
    for sc.Scan() {
        n++
        for _,p := range pats {
            if p.MatchString(sc.Text()) {
                out=append(out,FindingVuln{
                    File:path,Line:n,
                    Message:fmt.Sprintf("match:%s",p),
                    Severity:High,
                })
            }
        }
    }
    return out,sc.Err()
}
