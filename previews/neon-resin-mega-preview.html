use std::path::PathBuf;
use walkdir::WalkDir;

#[derive(Debug, Clone)]
pub enum Severity {
    Critical, High, Medium, Low,
}

#[derive(Debug)]
pub struct FindingVuln {
    pub file:     PathBuf,
    pub line:     u32,
    pub message:  String,
    pub severity: Severity,
}

pub trait Scannable {
    fn scan(&self) -> Vec<FindingVuln>;
    fn is_allowed(&self, ext: &str) -> bool;
}

const MAX_SIZE: u64 = 10_485_760;
const LDAP_RE: &str = r"LDAP://[\w\.]+";

pub fn scan_file<'a>(
    path:     &'a PathBuf,
    patterns: &[(&str, Severity)],
) -> Vec<FindingVuln> {
    let contents = std::fs::read_to_string(path)
        .unwrap_or_default();
    let mut findings: Vec<FindingVuln> = Vec::new();
    for (i, line) in contents.lines().enumerate() {
        for (pat, sev) in patterns {
            if line.contains(pat) {
                findings.push(FindingVuln {
                    file:     path.clone(),
                    line:     i as u32 + 1,
                    message:  format!("Match: {pat}"),
                    severity: sev.clone(),
                });
            }
        }
    }
    findings
}

from dataclasses import dataclass, field
from pathlib    import Path
import re

MAX_FINDINGS = 500
LDAP_RE      = re.compile(r"LDAP://[\w\.]+")
CRED_RE      = re.compile(r"password\s*=\s*\S+")
DEFAULT_SEV  = "medium"

class Severity:
    CRITICAL = "critical"
    HIGH     = "high"
    MEDIUM   = "medium"
    LOW      = "low"

@dataclass
class ScanResult:
    file_path:  Path
    line_no:    int
    pattern:   str
    severity:  str  = DEFAULT_SEV
    findings:  list = field(default_factory=list)

def scan_file(
    path:     Path,
    rules:    list[ScanResult],
    max_size: int = 10_485_760,
) -> list[ScanResult]:
    """Scan a file against all pattern rules."""
    if not path.exists(): return []
    results: list[ScanResult] = []
    text = path.read_text(encoding="utf-8")
    for n, line in enumerate(text.splitlines(), 1):
        for rule in rules:
            if re.search(rule.regex, line):
                results.append(ScanResult(
                    file_path=path, line_no=n,
                    pattern=rule.name, severity=rule.severity,
                ))
    return results[:MAX_FINDINGS]

# Install-Package.ps1 — winget wrapper
function Install-Package {
    [CmdletBinding()]
    param (
        [Parameter(Mandatory=$true)] [string] $Query,
        [Parameter()]               [switch] $Uninstall,
        [Parameter()]               [int]    $MaxWidth = 120
    )
    $raw = winget search --query $Query
    if (-not $raw) { Write-Warning "No results"; return }
    $packages = $raw | Select-Object -Skip 2 | ForEach-Object {
        $cols = $_ -split '\s{2,}', 4
        [PSCustomObject] @{
            Name = $cols[0] -replace '\u2026$',''
            Id   = $cols[1]; Ver = $cols[2]
        }
    }
    $packages | Format-Table -AutoSize
    $idx = [int](Read-Host "Pick") - 1
    if ($Uninstall) { winget uninstall --id $packages[$idx].Id
    } else { winget install --id $packages[$idx].Id }
}

package com.securelyscoped.scanner;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.regex.Pattern;

public enum Severity { CRITICAL, HIGH, MEDIUM, LOW }

public record FindingVuln(
    Path file, int line,
    String message, Severity severity
) {}

public interface Scannable {
    boolean isAllowed(String ext);
}

public class ScanEngine implements Scannable {
    private static final long    MAX     = 10_485_760L;
    private static final Pattern LDAP_RE = Pattern.compile("LDAP://[\\w\\.]+");
    private final Path root;

    public ScanEngine(Path root) { this.root = root; }

    @Override
    public boolean isAllowed(String ext) {
        return java.util.List.of("rs","py","java","go").contains(ext);
    }

    public void run() {
        try {
            Files.walk(root)
                 .filter(Files::isRegularFile)
                 .filter(p -> isAllowed(p.toString()))
                 .forEach(System.out::println);
        } catch (Exception e) {
            System.err.println("Error: " + e.getMessage());
        }
    }
}

// scanner.go — SecurelyScoped Go edition
package scanner

import (
    "bufio"; "fmt"; "os"; "regexp"
)

var (
    LDAPRegex = regexp.MustCompile(`LDAP://[\w\.]+`)
    CredRegex = regexp.MustCompile(`password\s*=\s*\S+`)
)

type Severity string
const (
    Critical Severity = "critical"
    High     Severity = "high"
    Medium   Severity = "medium"
    Low      Severity = "low"
)

type FindingVuln struct {
    File     string
    Line     int
    Message  string
    Severity Severity
}

func ScanFile(path string, pats []*regexp.Regexp) ([]FindingVuln, error) {
    f, err := os.Open(path)
    if err != nil { return nil, err }
    defer f.Close()
    var out []FindingVuln
    sc, n := bufio.NewScanner(f), 0
    for sc.Scan() {
        n++
        for _, p := range pats {
            if p.MatchString(sc.Text()) {
                out = append(out, FindingVuln{
                    File:path, Line:n,
                    Message: fmt.Sprintf("match: %s",p),
                    Severity:High,
                })
            }
        }
    }
    return out, sc.Err()
}
